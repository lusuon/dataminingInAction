ws = data['windspeed1']
pmax = data['power_max']

pdev = data['power_dev']

plt.rc('font', family='STXihei', size=15)
plt.scatter(ws,pmax,s=10,c='red',label='最大值')
plt.scatter(ws,pmin,s=10,c='black',label='最小值')
